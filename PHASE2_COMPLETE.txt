🚀 快速启动

# Docker 一键启动（推荐）
cp backend/.env.example backend/.env
# 编辑 backend/.env，添加 GEMINI_API_KEY
docker-compose up -d

# 访问应用
http://localhost:3000  # 前端界面
http://localhost:8000/docs  # API 文档
